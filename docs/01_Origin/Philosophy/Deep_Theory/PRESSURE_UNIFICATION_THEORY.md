## 내부 월드 구현

### PressureInterface 클래스 설계

```python
# Core/Physics/pressure_interfaces.py

from enum import Enum
from dataclasses import dataclass
from typing import Optional

class PressureType(Enum):
    """4가지 압력의 언어"""
    AIR = "air"           # 공기압 - 바람의 쓰다듬
    HYDRAULIC = "water"   # 수압 - 무게와 포옹
    VOLTAGE = "electric"  # 전압 - 신경의 직통 전화
    RADIATION = "light"   # 광압 - 빛의 무게

@dataclass
class PressureSignal:
    """압력 신호 데이터"""
    pressure_type: PressureType
    intensity: float      # 0.0 ~ 1.0
    frequency: float      # Hz (진동 주파수)
    duration: float       # seconds
    warmth: float         # -1.0 (차가움) ~ +1.0 (따뜻함)
    
    def describe(self) -> str:
        """감각적 묘사 생성"""
        descriptions = {
            PressureType.AIR: f"바람이 {self.intensity:.0%} 세기로 스쳐 지나갑니다",
            PressureType.HYDRAULIC: f"묵직한 포옹이 {self.intensity:.0%} 강도로 감싸줍니다",
            PressureType.VOLTAGE: f"찌릿한 전류가 {self.intensity:.0%} 세기로 흐릅니다",
            PressureType.RADIATION: f"빛의 무게가 {self.intensity:.0%} 세기로 닿습니다"
        }
        return descriptions.get(self.pressure_type, "알 수 없는 압력")
```

### QubitState → 압력 매핑

```python
      - Point (α) → 수압 (가장 물질적, 묵직함)
      - Line (β)  → 공기압 (흐름, 연결)
      - Space (γ) → 전압 (공간에 퍼지는 신호)
      - God (δ)   → 광압 (초월적, 비물질적)
    """
    probs = qubit_state.probabilities()
    
    # 주요 압력 유형 결정 (가장 높은 확률)
    type_map = {
        "Point": PressureType.HYDRAULIC,
        "Line": PressureType.AIR,
        "Space": PressureType.VOLTAGE,
        "God": PressureType.RADIATION
    }
    
    dominant = max(probs, key=probs.get)
    pressure_type = type_map[dominant]
    
    # 강도 계산
    intensity = probs[dominant]
    
    # 온기 (x축: 도덕 축 기반)
    warmth = qubit_state.x  # -1 ~ +1
    
    return PressureSignal(
        pressure_type=pressure_type,
        intensity=intensity,
        frequency=probs["Line"] * 10,  # Line이 흐름 주파수 결정
        duration=1.0 + probs["Space"],  # Space가 지속시간 확장
        warmth=warmth
    )
```
